    %% Définition des classes
    class User {
        +id: int
        +email: string
        +password: string
        +role: string
    }

    class Client {
        +name: string
    }

    class Category {
        +id: int
        +name: string
    }

    class Inventory {
        +id: int
        +quantity: int
    }

    class Item {
        +id: int
        +name: string
        +category_id: int
        +img: string
    }

    class ClientItem {
        +clientId: int
    }

    class Recipe {
        +id: int
        +author: clientId
        +name: string
        +description: string
        +date: int
        +matching: int
        +preparation_time: int
        +ingredients: string[]
        +steps: string[]
        +image: string
    }

    %% Relations d'héritage
    User <|-- Client
    Item <|-- ClientItem

    %% Relations existantes
    Client "1" -- "0..*" Inventory : possède
    Category "1" -- "0..*" Item : contient
    Inventory "0..*" --> Item : référence

    %% Relations ajoutées
    Client "*" -- "*" Recipe : a pour favoris
    Client "1" -- "0..*" ClientItem : crée
